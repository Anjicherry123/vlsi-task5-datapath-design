// File: my_rom.v
//==========================
`timescale 1ns / 1ps

module my_rom(
    input [1:0] addr,
    output reg [3:0] data_out
);

always @(*) begin
    case(addr)
        2'b00: data_out = 4'b0001;
        2'b01: data_out = 4'b0010;
        2'b10: data_out = 4'b0100;
        2'b11: data_out = 4'b1000;
    endcase
end

endmodule
