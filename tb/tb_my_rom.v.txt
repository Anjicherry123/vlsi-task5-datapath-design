`timescale 1ns / 1ps

module tb_my_rom;

reg [1:0] addr;
wire [3:0] data_out;

my_rom uut (
    .addr(addr),
    .data_out(data_out)
);

initial begin
    addr = 2'b00; #10;
    addr = 2'b01; #10;
    addr = 2'b10; #10;
    addr = 2'b11; #10;

    $finish;
end

endmodule
